\^{o}  = ô 
\"{o}  = ö 
\H{o}  = ő 
\~{o}  = õ 
\c{c}  = ç 
\k{a}  = ą 
\l{}   = ł 
\={o}  = ō 
\b{o}  = o
\.{o}  = ȯ 
\d{u}  = ụ 
\r{a}  = å 
\u{o}  = ŏ 
\v{s}  = š 
\t{oo} = o͡o

\^\i   = î
\"\i   = ï
